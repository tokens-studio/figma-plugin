export type CompareStateType<Metadata = null> = {
  tokens: Record<string, ImportToken[]>
  themes: ImportTheme[]
  metadata?: RemoteTokenStorageMetadata | Metadata
};
 
export type ImportTheme = ThemeObject & {
  importType?: 'NEW' | 'UPDATE' | 'REMOVE'
};
